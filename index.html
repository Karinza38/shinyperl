Objects with Moo

package Point;

use Moo;

has 'x' => isa => sub { m{\d+} };
has 'y' => isa => sub { m{\d+} };

sub describe {
  print "A point at ($x, $y)";
}

Point->new(4,2)->describe; # A point at (4,2)

use Object::Pad;

class Point {
   has $x = 0;
   has $y = 0;

   BUILD {
     ($x, $y) = @_;
   }

   method move ($dX, $dY) {
      $x += $dX;
      $y += $dY;
   }

   method describe {
      print "A point at ($x, $y)";
   }
}

Point->new(5,10)->describe; # A point at (5,10)
